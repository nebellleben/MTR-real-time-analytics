# Dataflow Flex Template Dockerfile
# Builds and pushes a Docker image for Google Container Registry
# Then launches the Dataflow job using the flex template

FROM python:3.11-slim

WORKDIR /app

# Copy requirements first ( for faster builds)
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Copy source code
COPY src/main.py .

# Set environment variables
ENV PROJECT_ID=de-zoomcamp-485516
ENV REGION=asia-east2
ENV PUBSUB_SUBSCRIPTION=mtr-arrivals-dataflow
ENV BIGQUERY_TABLE=mtr_analytics.raw_arrivals
ENV TEMP_LOCATION=gs://de-zoomcamp-485516-mtr-data-lake/temp
ENV STAGING_LOCATION=gs://de-zoomcamp-485516-mtr-data-lake/staging

# Run the pipeline
CMD ["python", "main.py"]
